6.1. Описание событий

 

#include <sys/siginfo.h>

struct sigevent ev;

SIGEV_NONE_INIT (&ev);
SIGEV_SIGNAL_INIT (&ev, signo);
SIGEV_SIGNAL_CODE_INIT (&ev, signo, value, code);
SIGEV_SIGNAL_THREAD_INIT (&ev, signo, value, code);
SIGEV_PULSE_INIT (&ev, coid, prio, code, value);
SIGEV_UNBLOCK_INIT (&ev);
SIGEV_INTR_INIT (&ev);
SIGEV_THREAD_INIT (&ev, func, param, 0);
