6.1. Описание событий

 

#include <sys/siginfo.h>

struct sigevent ev;

SIGEV_NONE          пустое событие
SIGEV_SIGNAL        простой сигнал
SIGEV_SIGNAL_CODE   сигнал со значением
SIGEV_SIGNAL_THREAD сигнал, нацеленный в нить
SIGEV_PULSE         импульс
SIGEV_UNBLOCK       разблокировка
SIGEV_INTR          прерывание
SIGEV_THREAD        создание нити
