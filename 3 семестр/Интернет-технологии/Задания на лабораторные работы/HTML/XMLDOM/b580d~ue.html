<script language="JavaScript">

function LoadDoc()
{ var xmldoc=new ActiveXObject("Microsoft.XMLDOM")
xmldoc.async=false;
xmldoc.load("samples.xml") // загружаем документ в память
output=" Овощи и не только: \n";
vegetables=xmldoc.getElementsByTagName("vegetables");
// вывод всех дочерних узлов тега <vegetables> </vegetables>
for(var i=0;i<vegetables.length;i++) output+=vegetables.item(i).text; document.all.textar.value=output;
// затем выведем их значения в "textarea"

output=" Овощи : \n";
vegetables=xmldoc.getElementsByTagName("vegetables") ;
product=vegetables.item(0).getElementsByTagName("product");
// вывод именно узла <product> </product> тега <vegetables> </vegetables>
for(var i=0;i<product.length;i++) output+=product.item(i).text+" ";
document.all.textat.value=output;
document.all.textar.value=output;
}
</script>

</head>
<body onLoad="LoadDoc()">
