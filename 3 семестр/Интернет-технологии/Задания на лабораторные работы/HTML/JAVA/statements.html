break [метка];

break используется для прекращения выполнения последовательности операторов

Вы можете использовать break в операторе выбора switch и в циклических конструкциях while, for, for...in или do...while.

При появлении break происходит переход в следующую за последовательностью операторов часть программы, если метка опущена. При наличии метки происходит переход в ту часть программы на которую эта метка указывает.

Следующий пример иллюстрирует использование break: 


function BreakTest(breakpoint){
   var i = 0;
   while (i < 100){
      if (i == breakpoint) break;
      i++;
   }
   return(i);
}

@cc_on

Activates conditional compilation support.

The @cc_on statement activates conditional compilation in the scripting engine.

It is strongly recommended that you use the @cc_on statement in a comment, so that browsers that do not support conditional compilation will accept your script as valid syntax: 

/*@cc_on*/
...
(remainder of script)

continue [метка];

continue используется для прекращения текущей итерации и начала новой итерации

Вы можете использовать continue в циклических конструкциях while, for, for...in или do...while.

При появлении continue происходит переход на следующую итерацию, если метка опущена. При наличии метки происходит переход в ту часть программы на которую эта метка указывает.

Следующий пример иллюстрирует использование continue: 


function skip5(){
  var s = "", i=0;
  while (i < 10){
    i++;
    if (i==5){
      continue;
    }
    s += i;
  }
  return(s);
}

Конструкция do...while используется для выполнения блока опеараторов хотя бы один раз и пока условие_продолжения итерации имеет значение истина.

Следующий код показывает использование do...while: 


//функция возвращает список: буква_диска название_диска,
//если диск читается
function GetDriveList()
{
  var fso, s, n, e, x;
  fso = new ActiveXObject("Scripting.FileSystemObject");
  e = new Enumerator(fso.Drives);
  s = "";
  do{
    x = e.item();
    s = s + x.DriveLetter;
    s += " - ";
    if (x.DriveType == 3)
      n = x.ShareName;
    else if (x.IsReady)
      n = x.VolumeName;
    else
      n = "[Drive not ready]";
    s +=  n + "<br>";
    e.moveNext();
  }while (!e.atEnd());
  return(s);
}
document.write(GetDriveList());

for(инициализация; условие_продолжения; инкремент) операторы

Конструкция for используется для циклического выполнения блока операторов, пока условие_продолжения имеет значение истина

"Инициализация" содержит операторы, задающие начальные значения переменным

"Инкремент" содержит операторы изменяющие значения переменных на каждой итерации.

Следующий пример демонстрирует использование for: 


var myarray = new Array();
for (i = 0; i < 10; i++){
  myarray[i] = i;
}

for(переменная in объект) операторы

Конструкция for...in используется для просмотра элементов объекта(массива)

"переменная" - идентификатор, в который передаются свойства объекта

Следующий пример демонстрирует использование конструкции for...in: 


function ForInDemo(){
  var a, key, s = "";
  a = {"a" : "Athens" , "b" : "Belgrade", "c" : "Cairo"}
  for (key in a){
     s += a[key] + "<BR>";
  }
  return(s);
}

Метка: операторы

Метка используется для перехода в какую-нибудь часть программы с помощью break или continue.

Следующий пример демонстрирует использование метки: 


function labelDemo(){
    var a = new Array();
    var i, j, s = "", s1 = "";
  Outer:
    for (i = 0; i < 5; i++){
      Inner:
        for (j = 0; j < 5; j++){
          if (j == 2)
            continue Inner;
          else
            a[i,j] = j + 1;
        }
    }
    for (i = 0; i < 5; i++){
      s = ""
      for (j = 0; j < 5; j++){
        s += a[i,j];
      }
      s1 += s + "\n";
    }
    return(s1)
}

@set @имя_переменной = терм

Creates variables used with conditional compilation statements.

Терм - Zero or more unary operators followed by a constant, conditional compilation variable, or parenthesized expression.

Numeric and Boolean variables are supported for conditional compilation. Strings are not. Variables created using @set are generally used in conditional compilation statements, but can be used anywhere in JavaScript code.

Следующий пример демонстрирует использование @set: 


@set @myvar1 = 12
@set @myvar2 = (@myvar1 * 20)
@set @myvar3 = @_jscript_version

The following operators are supported in parenthesized expressions:

  • ! ~
  • * / %
  • + -
  • << >> >>>
  • < <= > >=
  • == != === !==
  • & ^ |
  • && ||

If a variable is used before it has been defined, its value is NaN. NaN can be checked for using the @if statement: 

@if (@newVar != @newVar)
  ...

This works because NaN is the only value not equal to itself.

switch(условие){
   case метка: операторы
   default : операторы
}

Конструкция switch...case...defaultиспользуется для организации выбора из нескольких вариантов.

case можно использовать несколько раз. Если метка в сase совпадает с условием, заданным в switch, то выполняются все операторы, идущие после данной метки.

Если нет ни одной метки равной условию, описанному в switch, то выполняются операторы заданные в default

Следующий пример демонстрирует использование конструкции switch...case...default: 


//функция для подсчета среднего балла по массиву оценок
function sredball(){
   var strnumber = "55324235";
   var five=0,four=0,three=0,two=0;
   for (var i = 0; i < strnumber.length; i++){
      switch (strnumber.charAt(i)){
         case '5' : five++; break;
         case '4' : four++; break;
         case '3' : three++; break;
         default : two++; break;
      }
    }
    var sum = five + four + three + two;
    var sred = five * 5 + four * 4 + three * 3 + two * 2;
    return sred / sum;
}

try...catch используется для обработки ошибочных ситуаций.

try объединяет собой участок программы, в котором предположительно может возникнуть ошибочная ситуация, например деление на ноль. В try с помощью конструкции THROW выражение; можно генерировать собственное сообщение об ошибке.

В catch происходит обработка соотвествующей ошибки. Имя_ошибки - произвольный идентификатор содержащий генерируемое сообщение об ошибке.

Следующий пример иллюстрирует использование try...catch и throw: 


function TryCatchDemo(x){
  try {
    try {
    if (x == 0)
      throw "x equals zero";
    else
      throw "x does not equal zero";
    }
    catch(e){
      if (e == "x equals zero")
        return(e + " handled locally.");
      else
        throw e;
    }
  }
  catch(e){
    return(e + " handled higher up.");
  }
}
document.write(TryCatchDemo(0));
document.write(TryCatchDemo(1));

while(условие_продолжения) операторы

while используется для организации цикла, в котором итерации будут выполняться пока условие_продолжения есть истина, в противном случае операторы в while не выполняться ни разу.

Пример использования while: 


function BreakTest(breakpoint){
   var i = 0;
   while (i < 100){
     if (i == breakpoint) break;
     i++;
   }
   return(i);
}
