Создание объекта: new Enumerator(элементы)

Объект Enumerator, представляет собой коллекцию элементов.

Пример работы с объектом Enumerator: 


//функция выводит список: буква_диска название_диска,
//если диск читается
function ShowDriveList(){
  var fso, s, n, e, x;
  fso = new ActiveXObject("Scripting.FileSystemObject");
  e = new Enumerator(fso.Drives);
  s = "";
  for (;!e.atEnd();e.moveNext()){
      x = e.item();
      s = s + x.DriveLetter;
      s += " - ";
      if (x.DriveType == 3) n = x.ShareName;
      else if (x.IsReady) n = x.VolumeName;
      else n = "[Drive not ready]";
      s +=  n + "<br>";
  }
  return(s);
}
